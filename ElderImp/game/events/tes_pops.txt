namespace = tes_pops

tes_pops.1 = { #slaves resettled
	type = country_event
	title = "tes_pops.1.t"
	desc = "tes_pops.1.desc"
	picture = slaves_working
	
	left_portrait = root.current_ruler
	right_portrait = scope:tes_pops_1_target.governor
	
	goto_location = scope:tes_pops_1_target
	
	trigger = {
		war = no
		any_owned_province = {
			total_population >= 20
			num_of_slaves >= 10
		}
		any_owned_province = {
			num_of_slaves < 10
			region = root.capital_scope
		}
		var:slaves_policy >= 1
	}
	
	immediate = {
		random_owned_province = {
			limit = {
				total_population >= 20
				num_of_slaves >= 10
			}
			save_scope_as = tes_pops_1_origin
		}
		random_owned_province = {
			limit = {
				num_of_slaves < 10
				region = root.capital_scope
			}
			save_scope_as = tes_pops_1_target
		}
	}
	
	option = {
		name = "tes_pops.1.a"
		scope:tes_pops_1_origin = {
			random_pops_in_province = {
				limit = { pop_type = slaves }
				move_pop = scope:tes_pops_1_target
			}
		}
	}
}

tes_pops.2 = { #pops converted into fighters
	type = country_event
	title = "tes_pops.2.t"
	desc = "tes_pops.2.desc"
	picture = persia_marching
	
	left_portrait = root.current_ruler
	right_portrait = scope:tes_pops_2_target.governor
	
	goto_location = scope:tes_pops_2_target
	
	trigger = {
		any_owned_province = {
			total_population >= 4
			any_pops_in_province = {
				percent <= 0.5
				OR = {
					pop_type = fighters
					pop_type = mages
					pop_type = merchants
				}
			}
			any_pops_in_province = {
				NOR = {
					pop_type = fighters
					pop_type = mages
					pop_type = merchants
					pop_type = rogues
					pop_type = slaves
				}

			}
		}
		var:fighters_policy >= 2
	}
	
	immediate = {
		random_owned_province = {
			limit = {
				total_population >= 4
				any_pops_in_province = {
					percent <= 0.5
					OR = {
						pop_type = fighters
						pop_type = mages
						pop_type = merchants
					}
				}
				any_pops_in_province = {
					NOR = {
						pop_type = fighters
						pop_type = mages
						pop_type = merchants
						pop_type = rogues
						pop_type = slaves
					}
				}
			}
			weight = {
				modifier = {
					factor = 10
					has_province_modifier = fighters_guild_hall_lvl1
				}
			}
			random_pops_in_province = {
				limit = {
					NOR = {
						pop_type = fighters
						pop_type = mages
						pop_type = merchants
						pop_type = rogues
						pop_type = slaves
					}
				}
				weight = {
					modifier = {
						factor = 5
						OR = {
							pop_type = citizen
							pop_type = tribesmen
						}
					}
				}
				save_scope_as = tes_pops_2_target
			}
		}
	}
	
	option = {
		name = "tes_pops.2.a"
		scope:tes_pops_2_target = {
			set_pop_type = fighters
		}
	}
}
